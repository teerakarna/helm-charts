  # List proxies
  curl http://localhost:{{ .Values.service.apiPort }}/proxies

  # Add a latency toxic to a proxy
  curl -X POST http://localhost:{{ .Values.service.apiPort }}/proxies/<name>/toxics \
    -d '{"name":"latency","type":"latency","attributes":{"latency":100,"jitter":10}}'

  # Remove a toxic
  curl -X DELETE http://localhost:{{ .Values.service.apiPort }}/proxies/<name>/toxics/latency
