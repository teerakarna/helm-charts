echoserver is deployed and returns request details (headers, body, environment) on every HTTP request.

{{- if .Values.ingress.enabled }}
{{- range .Values.ingress.hosts }}
  {{- range .paths }}
  Ingress: http{{ if $.Values.ingress.tls }}s{{ end }}://{{ .host }}{{ .path }}
  {{- end }}
{{- end }}
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "echoserver.fullname" . }} -o jsonpath="{.spec.ports[0].nodePort}")
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  curl http://$NODE_IP:$NODE_PORT/
{{- else if eq .Values.service.type "LoadBalancer" }}
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "echoserver.fullname" . }} --template "{{ "{{ range (index .status.loadBalancer.ingress 0) }}{{ . }}{{ end }}" }}")
  # NOTE: It may take a few minutes for the LoadBalancer IP to become available.
  curl http://$SERVICE_IP:{{ .Values.service.port }}/
{{- else }}
  # Port-forward to access locally:
  kubectl --namespace {{ .Release.Namespace }} port-forward svc/{{ include "echoserver.fullname" . }} 8080:{{ .Values.service.port }}
  curl http://localhost:8080/
  curl http://localhost:8080/ -H "X-Custom-Header: test-value"
{{- end }}

  helm test {{ .Release.Name }} --namespace {{ .Release.Namespace }}
